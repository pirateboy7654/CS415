# Moderate I/O with /dev/urandom (output redirected to /dev/null)
echo "Starting Process 1 - CPU and I/O Test"
head -c 500000 /dev/urandom > /dev/null

# Moderate CPU with bc (calculates pi to 1000 decimal places)
echo "Starting Process 2 - CPU Test"
bc <<< "scale=1000; 4*a(1)"

# Light I/O with dd
echo "Starting Process 3 - I/O Test"
dd if=/dev/zero of=/dev/null bs=1K count=100

# Minimal CPU usage with sleep
echo "Starting Process 4 - Light CPU Test"
sleep 2
